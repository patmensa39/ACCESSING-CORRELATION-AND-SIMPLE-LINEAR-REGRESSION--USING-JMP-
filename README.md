### Fit group 1

Fit Group(
	Bivariate(
		Y( :Removal ),
		X( :OD ),
		Density Ellipse( 0.9, {Line Color( {212, 73, 88} )} ),
		SendToReport(
			Dispatch( {}, "Bivar Plot", FrameBox, {Frame Size( 413, 289 )} ),
			Dispatch(
				{},
				"Bivariate Normal Ellipse P=0.900",
				OutlineBox,
				{Close( 0 )}
			)
		)
	),
	Bivariate(
		Y( :Removal ),
		X( :ID ),
		Density Ellipse( 0.9, {Line Color( {212, 73, 88} )} ),
		SendToReport(
			Dispatch( {}, "Bivar Plot", FrameBox, {Frame Size( 406, 289 )} ),
			Dispatch(
				{},
				"Bivariate Normal Ellipse P=0.900",
				OutlineBox,
				{Close( 0 )}
			)
		)
	),
	<<{Arrange in Rows( 2 )}
)
